-----------------------------------------------------------------
-------- Typeorm
-----------------------------------------------------------------
## npm install @nestjs/typeorm typeorm sqlite3

// to use TypeOrm
1- open app.module.ts
2- add to imports []
  import {TypeOrm} from "@nestjs/typeorm";
  TypeOrmModule.forRoot({
      type: "sqlite",
      database: "db.sqlite",
      entities: [],
      synchronize: true
    }),

// to create entity 
// entity creates repository automatically for the object we are creating
1- create entity class "user.entity.ts"
2- connect it to the parent's module imports[] >> "user.module.ts"
  imports: [TypeOrmModule.forFeature([UserEntity])],
3- add it to entities[] in imports[] in "app.module.ts"
  entities: [UserEntity],


// SQLite extension: only one time 
// when running program and trying to open db.sqlite >> error
// add SQLite extension, open view >> Command palette and write SQLite >> choose open Database >> db.sqlite
// it will add "SQLITE EXPLORER" in the explorer window

-----------------------------------------------------------------
-------- cookies
-----------------------------------------------------------------

## npm install cookie-session @types/cookie-session

1- main.ts >> add the following code 

const cookieSession = require("cookie-session");
app.use(cookieSession({
    keys: ["sessionTempKey"]
  }));

2-users.controller.ts >> storing and using data inside the session object

@Get("/colors/:color")
    setColorToSession(@Param("color") color: string, @Session() session: any){
        session.color = color; // now the session{} looks like this >> session{color: 'redForExample'}
    }
    @Get("/colors")
    getLastColors(@Session() session: any){
        return session.color;
    }
